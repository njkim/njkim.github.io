CREATE TABLE final AS
with b02006_tract as (
	select b.geoid, b.geoid2, t.name, b.geodisplaylabel
	       as label, b.hd01_vd10 as Korean, t.geom
	from b02006 b join tl_2014_06_tract t
	on b.geoid2 = t.geoid::bigint)
select A.name, A.label, A.Korean, sum(B.Korean) as neighbor, A.geom
from b02006_tract A join b02006_tract B
on ST_intersects(A.geom, B.geom)
where A.label ilike '%Santa Clara%'
group by A.name, A.label, A.Korean, A.geom
order by A.name;
